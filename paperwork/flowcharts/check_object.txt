doObjectCalculations(int x, int y) {
 Obj o = currentBoard[x][y]
 
 if (o instance of Blank) {
  int n = 0;
  for (obj in getNearbyObjects(x, y) {
   n++;
   if (n > getBornPop()) {
    break;
   }
  }
  if (n == getBornPop()) {
   newboard[x][y] = new Person();
  }
 }

 if (o instance of Person) {
  int n = 0;
  for (obj in getNearbyObjects(x, y) {
   n++;
   if (n > getOverPop()) {
    break;
   }
  }
  if (n > getOverPop() || n < getUnderPop()) {
   newboard[x][y] = new Blank();
  }
 }

}
